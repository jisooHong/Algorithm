```SQL
# CAT 먼저 출력하기 
# HAVING 에는 집계함수만 사용하는게 좋음
SELECT ANIMAL_TYPE, COUNT(*) AS count
FROM ANIMAL_INS
WHERE ANIMAL_TYPE = "Cat" OR ANIMAL_TYPE = "Dog"
GROUP BY ANIMAL_TYPE
ORDER BY ANIMAL_TYPE;


SELECT NAME,COUNT(NAME)
FROM ANIMAL_INS
GROUP BY NAME
HAVING COUNT(NAME) >= 2
ORDER BY NAME;

# 여러 부등호 쓸 때는 AND로 이어주기
SELECT HOUR(DATETIME), COUNT(DATETIME)
FROM ANIMAL_OUTS
WHERE 9 <= HOUR(DATETIME) AND HOUR(DATETIME) <= 19
GROUP BY HOUR(DATETIME)
ORDER BY HOUR(DATETIME);

# 로컬 변수 선언은 SET으로, 그리고 = 대신 := 사용
SET @HOUR := -1;
SELECT (@HOUR := @HOUR + 1) AS HOUR, 
    (SELECT COUNT(*) FROM ANIMAL_OUTS WHERE HOUR(DATETIME) = @HOUR) COUNT
FROM ANIMAL_OUTS
WHERE @HOUR <= 22
ORDER BY @HOUR;
```
